###  1.求字符串出现数字和的最小值，会有负数
字符串格式：a-zA-Z±  
示例:abc12ss-123b  
最小值：1+2+（-123）= -120
ab0c12ss-123b–1ss+dd–g1g  
-120
```
# qb12-34bb-56aa-78
while True:  
    try:  
        s1 = input()  
  
        for i in s1:  
            if i.isalpha():  
                s1 = s1.replace(i," ")  
                s1 = s1.replace('+',' ')  
        s2 = ''  
 for i in s1:  
            if i == '-':  
                s2 = s2+' '+i  
            else:  
                s2 = s2+i  
        list1 = s2.split()  
        # print(list1)  
 list2 = []  
        for i in list1:  
            # print(i)  
 if ('-' in i) and (len(i) > 1):  
                list2.append(int(i))  
            elif '-' not in i:  
                for j in list(i):  
                    list2.append(int(j))  
            else:  
                continue  
 # print(list2)  
 print(sum(list2))  
    except:          
		break
```
